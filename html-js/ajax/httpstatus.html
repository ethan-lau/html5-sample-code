jQuery HTTP Response Handling

- http response generated by httpstat.us


function getResponse(code){
	var url = "http://httpstat.us/"+code;
	$.get(url, function(data) {
		alert(data);	// success
	}).error(function(e){
		alert(e.responseText); //error
	});
}
